import aisuite as ai
client = ai.Client()

# swap providers by changing one string
models = ["openai:gpt-4o", "anthropic:claude",
          "groq:llama-3"]

for m in models:
    r = client.chat.completions.create(
        model=m,
        messages=msgs,
    )
